Sub idade()

Dim anoNascimento As Integer, anoAtual As Integer, idadeAtual As Integer, idadeFutura As Integer

anoNascimento = InputBox("Digite um ano de nascimento: ")
anoAtual = InputBox("Digite o ano atual: ")

idadeAtual = (anoAtual - anoNascimento)
idadeFutura = (idadeAtual + 17)

MsgBox ("Você tem " & idadeAtual & " anos de idade em " & anoAtual & " e terá " & idadeFutura & " anos de idade em " & (anoAtual + 17))

End Sub
